# blog-v3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a hand-rolled static blog hosted on GitHub Pages with a Nothing-inspired monochrome dot-grid aesthetic to document an OSSU CS self-study journey.

**Architecture:** A Python build script (`build.py`) reads Markdown posts with YAML frontmatter and renders them through Jinja2 HTML templates into a `_site/` output directory. GitHub Pages serves from the `/docs` folder (build output copied there on publish). A `--watch` flag adds file watching + a local HTTP server for dev.

**Tech Stack:** Python 3, `markdown`, `python-frontmatter`, `Pygments`, `Jinja2`, `watchdog`, `pytest`

---

## File Map

| File | Responsibility |
|---|---|
| `build.py` | CLI entry point, build pipeline, watch mode, dev server |
| `templates/base.html` | Shared HTML shell: `<head>`, CSS vars, dot-grid body, nav |
| `templates/index.html` | Homepage: status strip + sidebar/feed layout |
| `templates/post.html` | Post page: header, reading column, TOC, prev/next nav |
| `templates/curriculum.html` | Full OSSU course map with progress bars |
| `static/style.css` | All CSS — tokens, layout, components, code blocks |
| `data/curriculum.json` | OSSU course list with progress per course |
| `posts/*.md` | Markdown source files with YAML frontmatter |
| `tests/test_build.py` | Unit tests for build script functions |
| `.gitignore` | Ignore `_site/`, `venv/`, `__pycache__/` |
| `requirements.txt` | Python deps |

```html

  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <link rel="stylesheet" href="{{ static_root }}style.css">
</head>
<body>
  <div class="page-wrap">
    <nav class="nav">
      <a href="{{ root }}index.html" class="nav-logo">TRINIDAD.LOG</a>
      <ul class="nav-links">
        <li><a href="{{ root }}index.html">Feed</a></li>
        <li><a href="{{ root }}curriculum.html">Curriculum</a></li>
        <li><a href="{{ root }}about.html">About</a></li>
      </ul>
    </nav>
    {% block content %}{% endblock %}
  </div>
</body>
</html>
```

```python
#!/usr/bin/env python3
"""Static site builder for blog-v3."""

import re
import os
import json
import shutil
import argparse
from datetime import datetime
from pathlib import Path

import frontmatter
import markdown
from jinja2 import Environment, FileSystemLoader
from pygments.formatters import HtmlFormatter


ROOT = Path(__file__).parent
POSTS_DIR = ROOT / "posts"
TEMPLATES_DIR = ROOT / "templates"
STATIC_DIR = ROOT / "static"
DATA_DIR = ROOT / "data"
OUTPUT_DIR = ROOT / "_site"

_md = markdown.Markdown(
    extensions=["fenced_code", "codehilite", "tables", "toc"],
    extension_configs={
        "codehilite": {"css_class": "highlight", "guess_lang": False},
        "toc": {"permalink": False},
    },
)


def derive_slug(filename: str) -> str:
    """'2026-05-12-memory-pointers.md' → 'memory-pointers'"""
    name = Path(filename).stem
    # strip leading YYYY-MM-DD- date prefix if present
    name = re.sub(r"^\d{4}-\d{2}-\d{2}-", "", name)
    return name


def parse_post(filepath: str) -> dict:
    """Parse a .md file with YAML frontmatter. Returns post dict."""
    post = frontmatter.load(filepath)
    filename = Path(filepath).name

    # reset markdown instance state between calls
    _md.reset()
    content = _md.convert(post.content)

    date_str = str(post.metadata["date"])
    date_fmt = datetime.strptime(date_str, "%Y-%m-%d").strftime("%Y.%m.%d")

    return {
        "title": post.metadata["title"],
        "date": date_str,
        "date_fmt": date_fmt,
        "course": post.metadata["course"],
        "week": post.metadata.get("week"),
        "tags": post.metadata.get("tags", []),
        "excerpt": post.metadata["excerpt"],
        "slug": derive_slug(filename),
        "content": content,
    }


def extract_toc(html: str) -> list:
    """Return TOC items from rendered HTML. Empty list if fewer than 3 h2 tags."""
    matches = re.findall(r'<h2[^>]*id=["\']([^"\']+)["\'][^>]*>(.*?)</h2>', html, re.IGNORECASE)
    if len(matches) < 3:
        return []
    return [{"anchor": anchor, "text": re.sub(r"<[^>]+>", "", text)} for anchor, text in matches]
```

```python
def load_curriculum() -> list:
    with open(DATA_DIR / "curriculum.json") as f:
        return json.load(f)["courses"]


def compute_stats(courses: list) -> dict:
    total = len(courses)
    completed = sum(1 for c in courses if c["status"] == "complete")
    overall_pct = round(sum(c["progress"] for c in courses) / total) if total else 0
    return {
        "overall_pct": overall_pct,
        "completed_courses": completed,
        "total_courses": total,
    }


def _jinja_env() -> Environment:
    env = Environment(loader=FileSystemLoader(str(TEMPLATES_DIR)))
    return env


def _get_current_course(courses: list) -> dict:
    for c in courses:
        if c["status"] == "in_progress":
            return c
    return courses[0]


def build_all():
    # 1. clean + create output dirs
    if OUTPUT_DIR.exists():
        shutil.rmtree(OUTPUT_DIR)
    OUTPUT_DIR.mkdir(parents=True)
    (OUTPUT_DIR / "posts").mkdir()

    # 2. copy static assets
    if STATIC_DIR.exists():
        shutil.copytree(STATIC_DIR, OUTPUT_DIR / "static")

    # 3. parse all posts
    md_files = sorted(POSTS_DIR.glob("*.md"), reverse=True)
    posts = [parse_post(str(f)) for f in md_files if not f.name.startswith(".")]

    # 4. load curriculum + compute stats
    courses = load_curriculum()
    stats = compute_stats(courses)
    current = _get_current_course(courses)

    # 5. count posts per course
    post_counts = {}
    for p in posts:
        key = p["course"]
        post_counts[key] = post_counts.get(key, 0) + 1
    for c in courses:
        c["post_count"] = post_counts.get(c["name"], 0)

    # 6. group courses by category
    categories: dict = {}
    for c in courses:
        categories.setdefault(c["category"], []).append(c)

    env = _jinja_env()

    # 7. render each post
    post_tmpl = env.get_template("post.html")
    for i, post in enumerate(posts):
        toc = extract_toc(post["content"])
        prev_post = posts[i + 1] if i + 1 < len(posts) else None
        next_post = posts[i - 1] if i > 0 else None
        html = post_tmpl.render(
            post=post,
            toc=toc,
            prev_post=prev_post,
            next_post=next_post,
            root="../",
            static_root="../static/",
        )
        (OUTPUT_DIR / "posts" / f"{post['slug']}.html").write_text(html)

    # 8. render homepage
    index_tmpl = env.get_template("index.html")
    html = index_tmpl.render(
        posts=posts,
        sidebar_courses=courses,
        current_course=current["name"],
        current_week=_latest_week(posts, current["name"]),
        root="",
        static_root="static/",
        **stats,
    )
    (OUTPUT_DIR / "index.html").write_text(html)

    # 9. render curriculum page
    curr_tmpl = env.get_template("curriculum.html")
    html = curr_tmpl.render(
        categories=categories,
        root="",
        static_root="static/",
        **stats,
    )
    (OUTPUT_DIR / "curriculum.html").write_text(html)

    print(f"Built {len(posts)} posts → {OUTPUT_DIR}")


def _latest_week(posts: list, course_name: str) -> int:
    weeks = [p["week"] for p in posts if p["course"] == course_name and p["week"]]
    return max(weeks) if weeks else 1
```

```python
def watch_and_serve(port: int = 8000):
    """Rebuild on file changes, serve on localhost:port."""
    from watchdog.observers import Observer
    from watchdog.events import FileSystemEventHandler
    import threading
    import http.server

    build_all()

    class RebuildHandler(FileSystemEventHandler):
        def on_any_event(self, event):
            if event.is_directory:
                return
            if any(event.src_path.endswith(ext) for ext in (".md", ".html", ".css", ".json")):
                print(f"Changed: {event.src_path} — rebuilding…")
                try:
                    build_all()
                except Exception as e:
                    print(f"Build error: {e}")

    observer = Observer()
    for watch_dir in [POSTS_DIR, TEMPLATES_DIR, STATIC_DIR, DATA_DIR]:
        if watch_dir.exists():
            observer.schedule(RebuildHandler(), str(watch_dir), recursive=True)
    observer.start()
    print(f"Watching for changes. Serving at http://localhost:{port}")

    os.chdir(OUTPUT_DIR)
    handler = http.server.SimpleHTTPRequestHandler
    handler.log_message = lambda *a: None  # silence request logs
    httpd = http.server.HTTPServer(("", port), handler)
    try:
        httpd.serve_forever()
    except KeyboardInterrupt:
        observer.stop()
    observer.join()


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="blog-v3 builder")
    parser.add_argument("--watch", action="store_true", help="watch + serve")
    parser.add_argument("--port", type=int, default=8000)
    args = parser.parse_args()

    if args.watch:
        watch_and_serve(args.port)
    else:
        build_all()
```
